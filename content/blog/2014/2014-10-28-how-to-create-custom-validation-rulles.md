```php
$form->addText('domain', 'Domain:')
	->addRule(DomainsRules::DOMAIN, 'fill domain in format domain.tld');

$form->addText('user', 'User:')
	->addRule(DomainsRules::USER_IN_DOMAIN, 'user is not in domain nette.org', 'nette.org');
```
